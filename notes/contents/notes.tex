\section{Objective}
Let $g : X\times X \to \bbR$ be some function, and for any $t\geq 0$, let
\begin{equation}
    \begin{split}
        \label{eq:criteria}
        E(t)&\coloneq \bbE_{X_t\sim \pi_t(\dd x), Y_{t+1}\sim M_{t+1}(X_t, \dd y)}[g(X_t, Y_{t+1})]\\
        &= \bbE_{X_t\sim \pi_t(\dd x)}[\bbE_{Y_{t+1}\sim M_{t+1}(X_t,\dd y)}[g(X_t, Y_{t+1})\mid X_t]]
    \end{split}
\end{equation}
Given $E(t)$ monitors the quality of the proposal $M_{t+1}$, we aim to construct $M_{t+1}$ so to optimize $E(t)$.


\section{Adaptive Waste-Free SMC}
At iteration $1\leq t\leq T$, for the particle $m$ and two consecutive particles with same ancestor $1\leq m \leq M$, $2\leq p\leq P$,
\begin{equation}
    \label{eq:current_samples}
    (\tilde{X}_t^{m, p-1}, \tilde{X}_t^{m, p})
    \sim \pi_{t-1}(\dd x_t[p-1]) \otimes M_t(x_t[p-1], \dd x_t[p]).
\end{equation}
\begin{proof}
    By construction, $(\tilde{X}_t^{m, 1}, \tilde{X}_t^{m,2},\ldots, \tilde{X}_t^{m,p})\sim \pi_{t-1}(\dd x_t[1])\prod_{p=2}^{P} M_t(x_t[p-1], \dd x_t[p])$, and the result is true for $p=2$.
    For $2<p\leq P$, for any test function $g$, since $\pi_{t-1}M_t = \pi_{t-1}$
    \begin{equation}
        \begin{split}
            &\int g(x_t[p-1], x_t[p]) \pi_{t-1}(\dd x_t[1])M_t(x_t[1], \dd x_t[2])\ldots M_t(x_t[p-1], \dd x_t[p])\\
            =&\int g(x_t[p-1], x_t[p]) \pi_{t-1}(x_t[2]) \ldots M_t(x_t[2], x_t[3])\ldots M_t(x_t[p-1], \dd x_t[p])\\
            =&\int g(x_t[p-1], x_t[p]) \pi_{t-1}(x_t[p-1]) M_t(x_t[p-1], \dd x_t[p]).
        \end{split}
    \end{equation}
\end{proof}

Let $t\geq 1$ and $\hat{\pi}_t\coloneq \sum_{n=1}^{N}\delta_{X_t^n}W_t^n$, $N=MP$, where $(X_t^n)_{1\leq n\leq N}$ are the gathered particles $\tilde{X}_{t}^{m,p}$ for $1\leq p\leq P$, $1\leq m \leq M$, then $\hat{\pi}_t$ is an estimate of $\pi_t$.
Let $(M^{\theta}_{t+1})_{\theta\in \Theta}$ be a collection of Markov kernels leaving $\pi_t$ invariant and let $E(t,\theta)$ be~\eqref{eq:criteria} with the dependency in $\theta$ made explicit.
We aim to construct an estimate of $E(t, \theta)$, let it be $\hat{E}(t, \theta)$, using the samples of iteration $t$~\eqref{eq:current_samples} then compute
\begin{equation}
    \label{eq:maximisation_procedure}
    \hat{\theta}_{t+1}^{*} = \textup{argmax}_{\theta\in \Theta} \hat{E}(t, \theta).
\end{equation}
We propose, for any $x\in X$, to perform importance sampling from $M_t(x, \dd y)$ to $M_{t+1}^{\theta}(x, \dd y)$ (assuming the density ratio $\dd M_{t+1}^{\theta}(x, y)/\dd M_t(x, y)$ exists) to estimate the inner expectation $\bbE_{Y_{t+1}\sim M_{t+1}^{\theta}(x, \dd y)}$, and to estimate the outer expectation $\bbE_{X_t\sim \pi_t(\dd x)}$ by the empirical reweighted mean over the particles.
For the inner expectation, for all $x\in X$, $\theta\in \Theta$
\begin{equation}
    \bbE_{Y_{t+1}\sim M_{t+1}^{\theta}(x, \dd y)}[g(x, Y_{t+1})] = \bbE_{Y_{t}\sim M_t(x, \dd y)} [g(x, Y_{t})  w_{t+1}^{\theta}(x, Y_t)],
\end{equation}
with $w_{t+1}^{\theta}(x, y) \coloneq \dd M_{t+1}^{\theta}(x, y)/\dd M_t(x, y)$.
Replacing this expectation by the unbiased estimate obtained with one-sample, and estimating the outer expectation using the reweighted mean over the particles, we obtain the unbiased estimate
\begin{equation}
    \hat{E}(t, \theta) \coloneq \frac{1}{N}\sum_{1\leq p\leq P, \\
    1\leq m\leq M} W_t^{m, p-1}g(\tilde{X}_t^{m, p-1}, \tilde{X}_t^{m, p}) w_{t+1}^{\theta}(\tilde{X}_t^{m,p-1}, \tilde{X}_t^{m,p}).
\end{equation}

To avoid importance sampling issue, we can consider restrictive objectives or penalised objectives to replace~\eqref{eq:maximisation_procedure}, taking into account the effective sample size, the variance of the weights, or replacing $\Theta$ by a neighbourhood centered around $\hat{\theta}_t$.

\paragraph{MH kernel}
At time $t\geq 1$, let $Q_t(x, \dd y)$ be some proposal distribution conditionally on $x\in X$.
The associated kernel is
\begin{equation}
    M_t(x, \dd y) = \delta_x (\dd y) \left(\int (1-a_t(x, y))Q_t(x, \dd y)\right) + a_t(x, y)Q_t(x, \dd y),
\end{equation}
with $a_t(x, y) = 1\wedge \frac{\pi_{t-1}(y)Q_t(y, x)}{\pi_{t-1}(x)Q_t(x, y)}$ the Metropolis-Hasting ratio.
We use a Rao-Blackwellised estimate for $E$.
Indeed, assume $g(x, x) = 0$, then for any $t\geq 0$, we have
\begin{equation}
    \label{eq:esjd_s_specialized}
    E(t) = \bbE_{X_t\sim \pi_t(\dd x)} \left[\bbE_{\bar{X}_{t+1}\sim Q_{t+1}(X_t, \dd \bar{x})}[a_{t+1}(X_t, \bar{X}_{t+1}) g(X_t, \bar{X}_{t+1})\mid X_t]\right]
\end{equation}
using proposals $\bar{X}_{t+1}\sim Q_{t+1}(X_t, \dd \bar{x})$ before the acceptation-rejection procedure.
For any $t\geq 1$,  $2\leq p \leq P$, $1\leq m\leq M$, let $\bar{X}_t^{m, p} \sim Q_t(\tilde{X}_t^{m,p-1}, \cdot)$ and $\bar{X}_t^{m,1} = X_{t-1}^{A_t^m}$ where $A_t^m\sim \textup{resample}(M, W_{t-1}^{1:N})$ be the proposed particles before acceptation-rejection step at iteration $t$.
Then an estimate of $E(t, \theta)$ is given by
\begin{equation}
    \begin{split}
        \label{eq:esjd_estimate}
        &\hat{E}^{\theta}_t = \frac{1}{M(P-1)}\sum_{2\leq p\leq P, 1\leq m\leq M}^{P}\bar{W}_t^{m,p-1}  \norm{\tilde{X}_t^{m,p-1}-\bar{X}_{t}^{m,p}}^2 a_{t+1}^{\theta}(\tilde{X}_t^{m,p-1}, \bar{X}_{t}^{m,p}) w_{t+1}^{\theta,'}(\tilde{X}_t^{m,p-1}, \bar{X}_{t}^{m,p}),\\
        &w_{t+1}^{\theta,'}(x, y) \coloneq \frac{Q_{t+1}^{\theta}(x, y)}{Q_t(x,y)},\,\, a_{t+1}^{\theta}(x,y) \coloneq 1 \wedge \frac{\pi_t(y)Q_{t+1}^{\theta}(y, x)}{\pi_t(x) Q_{t+1}^{\theta}(x, y)},\,\, \bar{W}_t^{m,p-1}\propto W_{t}^{m, p-1}, \sum_{p=2}^{P}\sum_{m=1}^{M}\bar{W}_t^{m,p-1} = 1.
    \end{split}
\end{equation}
While for calibrating $M_{1}^{\theta}$, i.e., estimating the optimal $\theta_1$, we estimate~\eqref{eq:esjd_s_specialized} using samples from $\nu$
\begin{equation}
    \begin{split}
        &\hat{E}^{\theta}_{0} = \frac{1}{N} \sum_{n=1}^{N} W_0^n \norm{X_0^n - \bar{X}_1^n}^2 a_1^{\theta}(X_0^n, \bar{X}_1^n),\,\, \bar{X}_1^n \sim Q_1^{\theta_0}(X_0^n, \dd \bar{x}),\,\, a_{1}^{\theta}(x,y) \coloneq 1 \wedge \frac{\pi_0(y)Q_{1}^{\theta}(y, x)}{\pi_0(x) Q_{1}^{\theta}(x, y)}.
    \end{split}
\end{equation}
In the case where $Q(x, y)=Q(y, x)$, the previous function is constant and thus cannot be optimised with respect to $\theta_1$.
The estimate~\eqref{eq:esjd_estimate} is (almost) free to compute since all the involved terms are already computed during the procedure at iteration $t$, except $Q_{t+1}^{\theta}(\bar{X}_t^{m,p}, \tilde{X}_t^{m, p-1})$ and $Q_{t+1}^{\theta}(\tilde{X}_t^{m, p-1}, \bar{X}_t^{m,p})$.
\begin{algorithm}[!ht]
    \label{alg:adaptative_waste_free_smc_rwmh}
    \DontPrintSemicolon
    \caption{Adaptative Waste Free SMC with MH proposals}
    \SetKwInOut{Input}{input}
    \SetKwInOut{Output}{output}
    \

    \Input{Integers $M, P\geq 1$, $N\gets MP$, a FK model (initial distribution $\nu$, functions $G_t$, parametric MH proposals $(Q_t^{\theta})_{\theta\in \Theta}$), a criterion $g$}
    \For{$t\gets 0$ \KwTo $T$}{
        \If{$t=0$}{
            \For{$n\gets 1$ \KwTo $N$}{
                $X_0^n\sim \nu(\dd x_0)$\;
                $\bar{X}_1^n\sim Q_1^{\theta_0}(X_0^n,\dd\bar{x})$\;
                $g_{1}^n \gets \frac{\pi_0(\bar{X}_1^n)}{\pi_0(X_0^n)}$\;
                $d_{1}^n \gets g(X_0^n, \bar{X}_1^n)$\;
            }
        }\Else {
            $A_t^{1:M}\sim \text{resample}(M, W_{t-1}^{1:N})$\;
            \For{$m\gets 1$ \KwTo $M$}{
                $\tilde{X_t}^{m, 1}\gets X_{t-1}^{A_{t}^m}$\;
                \For{$p\gets 2$ \KwTo $P$}{
                    $\bar{X_t}^{m, p}\sim Q_t^{\theta_t}(\tilde{X}_t^{m, p-1}, \dd x_t)$\;
                    $g_{t+1}^{m,p}\gets \frac{\pi_t(\bar{X}_t^{m,p})}{\pi_t(\tilde{X}_t^{m, p-1})}$\;
                    $d_{t+1}^{m,p}\gets g(\tilde{X}_t^{m,p-1},\bar{X}_t^{m,p})$\;
                    $q_{t+1}^{m,p}\gets Q_t^{\theta_t}(\tilde{X}_t^{m,p-1}, \bar{X}_t^{m,p})$\;
                    $U\sim \calU[0, 1]$\;
                    \If{$U< 1 \wedge \frac{\pi_{t-1}(\bar{X_t}^{m, p})Q_t^{\theta_t}(\bar{X_t}^{m, p}, \tilde{X}_t^{m,p-1})}{\pi_{t-1}(\tilde{X}_t^{m, p-1})\textcolor{green}{q_{t+1}^{m,p}}}$}{
                        $\tilde{X_t}^{m, p}\gets \bar{X_t}^{m,p}$\;
                    }\Else {
                        $\tilde{X_t}^{m, p}\gets \tilde{X_t}^{m, p-1}$\;
                    }
                }

            }
        }
        \For{$n\gets 1$ \KwTo $N$}
        {
            $w_t^n\gets G_t(X_t^n)$\;
        }
        \For{$n\gets 1$ \KwTo $N$}
        {
            $W_t^n \gets w_t^n / \sum_{n=1}^{N}w_t^n$\;
        }
        \If{$t=0$}{
            $\theta_1\gets \text{argmax}_{\theta\in \Theta}\sum_{n=1}^{N} W_0^n d_1^n \times \left(1\wedge g_{1}^n \frac{Q_1^{\theta}(\bar{X}_1^n, X_0^n)}{Q_1^{\theta}(X_0^n, \bar{X}_1^n)}\right)$\;
        }\Else {
            $\theta_{t+1}\gets \text{argmax}_{\theta\in \Theta}\sum_{2\leq p\leq P, 1\leq m\leq M}\bar{W}_t^{m, p-1} d_{t+1}^{m,p} \left(1 \wedge g_{t+1}^{m,p} \frac{Q_{t+1}^{\theta}(\bar{X}_t^{m, p}, \tilde{X}_t^{m, p-1})}{Q_{t+1}^{\theta}(\tilde{X}_t^{m, p-1}, \bar{X}_t^{m, p})}\right)\times \frac{Q^{\theta}_{t+1}(\tilde{X}_t^{m,p-1}, \bar{X}_t^{m,p})}{q_{t+1}^{m,p}}$\;
        }
    }
\end{algorithm}

\paragraph{Random Walk: optimal scaling}
Proposals of the form $y=x + l\sqrt{n^{-\gamma}}Z$, where $Z\sim \calN(0, I_n)$.
Assume $\pi = f^{\otimes n}$,~\citet{robertsgalman97} show that $\gamma=1$, and the optimal asymptotic scaling $l$ satisfies
\begin{equation}
    l = 2.38/\sqrt{I}, I = \bbE[ \log(f)'(X)^2],
\end{equation}
and yields an acceptance rate of $2\phi(-\sqrt{I}l/2)\approx 0.234$

\subparagraph{Experiment}
Let $\Theta = \bbR^{+}$, let $\theta\coloneq l$.
Let $\pi_0 = \calN(0, I_n)$, and let $L$ be such that $L\pi_0 = \calN(\mu, \sigma^2 I_n)$.
Fix $T\geq 1$, and let $\pi_t = L^{\gamma_t}\pi_0$ with $\gamma_t = t/T$, for $0\leq t\leq T$.
We consider the Adaptative RW proposal\yvann{find ref}, at iteration $t\geq 1$,
\begin{equation}
    \label{eq:rwmh_adaptative_smc}
    y \sim \calN(x, l^2/n \hat{\Sigma}_{t-1}),
\end{equation}
where $\hat{\Sigma}_{t-1}$ is the estimated covariance matrix of the target distribution $\pi_{t-1}$ given by
\begin{equation}
    \hat{\Sigma}'_{t-1}=(N-1)^{-1}\sum_{1\leq i\leq N}W_{t-1}(X_{t-1}^{n}-\hat{\mu}_{t-1})^{\top}(X_{t-1}^{n}-\hat{\mu}_{t-1}), \hat{\mu}_{t-1} = N^{-1}\sum_{1\leq i\leq N} W_{t-1}^{n}X_{t-1}^n,
\end{equation}
The ESJD criterion is obtained with $g(x, y) = \norm{x-y}^2$, and the corresponding estimator of~\eqref{eq:criteria} given by~\eqref{eq:esjd_estimate}.
Since all the intermediary distributions $\pi_t$ are products of identical unidimensional Gaussian distributions, the maximum of the exact ESJD is achieved for the optimal scaling $l=2.38$ in~\eqref{eq:rwmh_adaptative_smc} when $n\to\infty$.
Indeed, let $\pi_t = \calN(\mu_t, \sigma^2_t I_n)$, then a computation shows that $I_t= 1/\sigma^2_t$, thus $y = x+2.38\sigma_t/\sqrt{n} Z$, with $Z\sim \calN(0, I_n)$ is the optimal proposal at iteration $t+1$ targeting $\pi_t$.
Since $\hat{\Sigma}_t\to \sigma_t^2 I_n$, a.s., as $N$ goes to $\infty$, the proposal~\eqref{eq:rwmh_adaptative_smc} should be close in law to the optimal proposal when $l=2.38$.
We aim to estimate the optimal scaling $l$ obtained via maximising~\eqref{eq:esjd_estimate}.
As $n$ goes to $\infty$ and the number of particles $N=MP$ goes to $\infty$, the optimal scaling $\hat{l}_{t}$ should converge to $2.38$, for all $1\leq t\leq T$.
More generally, one can replace $\hat{\Sigma}_{t-1}$ in~\eqref{eq:rwmh_adaptative_smc} by
\begin{equation}
    \hat{\Sigma}_{t-1} = N^{-1}\sum_{1\leq i\leq N} W_{t-1}^{i} \nabla \log \pi_{t-1}(X_{t-1}^{i}).
\end{equation}

\paragraph{Metropolis Adjusted Langevin Algorithm}
Proposals of the form $y=x + l^2n^{-\gamma}\nabla \log\pi(x) /2 + l\sqrt{n^{-\gamma}}Z$, with $Z\sim\calN(0, I_n)$.
For i.i.d target distributions,~\citet{garethrosenthal98} show that $\gamma=1/3$ and the optimal asymptotic scaling $l$ satisfies
\begin{equation}
    2\phi(-Jl^3/2)\approx 0.574, J^2= \frac{1}{48}\bbE[5\log(f)'''(X)^2 - 3\log(f)''(X)^3].
\end{equation}

\subparagraph{Experiment}
Similarly to the previous example, we consider $\pi_t = \calN(\mu_t, \sigma_t^2 I_n)$.
A computation shows that $J^{-2}_t = 16\sigma_t^6$. Let $l'$ be such $2\phi(-J_tl^{',3}/2)\approx 0.574$, i.e., $l'\approx 1.04J_t^{-1/3} = 1.650\sigma_t$.
Thus, we consider the following adaptative MALA proposal, at iteration $t\geq 1$
\begin{equation}
    \label{eq:mala_adaptative_smc}
    y \sim \calN(x + l^2 n^{-1/3} \textup{diag}\hat{\Sigma}_{t-1} \nabla \log\pi_{t}(x)/2, l^2n^{-1/3}\hat{\Sigma}_{t-1}),
\end{equation}
where $\hat{\Sigma}_{t-1}$ is the empirical covariance estimate of $\pi_{t-1}$ computed using particles and weights of iteration $t-1$.
As $N$ goes to $\infty$, the optimal scaling $\hat{l}_{t}$ should be close to $\approx 1.650$, for all $1\leq t\leq T$.
Again, instead of approximating expectations under $\pi_{t-1}$ of quantities involving $\pi_{t-1}$, we can directly estimate expectations of quantities inolving $\log \pi_{t}$ using samples $(X_{t-1}^{i})$ and IS.
In the general case where $\pi = f^{\otimes n}$, let $\hat{J^2_t}$ be the unbiased estimate of $J_t^2$ given by
\begin{equation}
    \hat{J_t^2} = 48^{-1}N^{-1}\sum_{1\leq i\leq N}W_{t-1}^{i}(X_{t-1}^{i})\left(5 (\nabla^3 \log \pi_t(X_{t-1}^{i}))^{2} - 3(\nabla^{2} \log \pi_t(X_{t-1}^{i})^{\top})^{3})\right)^2,
\end{equation}
where the derivatives are taken component-wise.
Let $\hat{l}^{'}\coloneq 1.04(\hat{J^2_t})^{-1/6}$, the proposal is then
\begin{equation}
    y\sim \calN(x + \hat{l}^{',2} n^{-3} \nabla \log \pi_t(x)/2, \hat{l}^{', 2}n^{-1/3}).
\end{equation}

\paragraph{RWMH kernel: Autoregressive proposal}
\label{subsec:ar_proposal}
Let $C \succ 0$, $\nu(\dd x)$ some base probability measure on $X$, e.g., $\nu(\dd x) = \calN(0, C)$.
Let $\Theta = [-1, 1]$, and let for any $t\geq 1$, $x\in X$, $\rho \in \Theta$, $Q_s^{\rho}(x, \dd y) = \calN(\dd y \mid \rho x, (1-\rho^2)C)$.
At iteration $t\geq 1$, let us denote by $\rho_1, \rho_2, \ldots \rho_{t} \in \Theta$ the sequence of autoregressive parameters for the proposals $Q_1, Q_2, \ldots, Q_{t}$.

\paragraph{Taking into account higher-order autocorrelations}
Let $\phi$ be some target function for which we are interested in estimating $\pi_0(\phi), \ldots, \pi_T(\phi)$.
Let $t\geq 0$, let us choose $\theta^{t+1}$ so to maximise the efficiency for the next MH chain with $\pi_t$ as invariant measure
\begin{equation}
    \textup{eff}(t, \nu)\coloneq P^{-1}\textup{Var}_{X\sim \pi_{t}}[\phi(X)] / \bbE[(\hat{\phi}_t(\nu)-\pi_t[\phi])^2],
\end{equation}
where
\begin{equation}
    \hat{\phi}_t(\nu)\coloneq \frac{1}{P}\sum_{1\leq p\leq P} \phi(X_{t+1}^{*,p}),\indent (X_{t+1}^{*,1},\ldots, X_{t+1}^{*,P})\sim \pi_t(\dd x_{t+1}[1])\prod_{p=2}^{P}M_{t+1}^{\nu}(x_{t+1}^{*}[p-1], \dd x_{t+1}^{*}[p]).
\end{equation}
\begin{proposition}
    \begin{equation}
        \begin{split}
            \textup{eff}(t,\nu) = \left(1 + \frac{2}{P \rho_{t}(0)}\sum_{l=1}^{P-1}(P-l)\rho_{t}(l)\right)^{-1},
        \end{split}
    \end{equation}
    where $\rho_{t}(l)\coloneq \bbE[(\phi(X^{*,1}_{t+1})-\pi_t(\phi))(\phi(X^{*,1+l}_{t+1})-\pi_t(\phi))]$.
    \begin{proof}
        \begin{equation}
            \begin{split}
                \bbE[(\hat{\phi}_t(\nu)-\pi_t(\phi))^2] &= P^{-1}\textup{Var}_{X\sim \pi_t}(\phi(X)) + 2P^{-2} \sum_{1\leq p< p'\leq P}\bbE[(\phi(X_{t+1}^{*,p})-\pi_t(\phi))(\phi(X_{t+1}^{*,p'})-\pi_t(\phi))]\\
                &= P^{-1}\textup{Var}_{X\sim \pi_t}(\phi(X)) + 2P^{-2}\sum_{1\leq l\leq P-1}(P-l)\rho_t(l).
            \end{split}
        \end{equation}
        Taking the inverse and dividing by $P^{-1}\textup{Var}_{X\sim \pi_t}[\phi(X)]$ yields the result.
    \end{proof}
\end{proposition}

            &\times( a_{t+1}^{\nu}(\tilde{X}_{t}^{m,p}, \bar{X}_{t}^{m,p+2}) \phi(\bar{X}_{t}^{m,p+2}) + (1-a_{t+1}^{\nu}(\tilde{X}_{t}^{m,p}, \bar{X}_{t}^{m,p+2}))\phi(\tilde{X}_{t}^{m,p}))
            \}\\
            &-\hat{\pi}_t(\phi)^2
        \end{split}
    \end{equation}
    \begin{proof}
        \begin{equation}
            \label{eq:exp_order2}
            \begin{split}
                \bbE[\phi(X^{*,p}_{t+1})\phi(X^{*,p+2}_{t+1})]&=\bbE_{X_{t+1}^{*,p}\sim \pi_t}[\phi(X^{*,p}_{t+1})\\
                &\times \bbE_{X_{t+1}^{*,p+1}\sim M_{t+1}(X_{t+1}^{*,p}, \dd x_{t+1}^{*,p+1})}[\bbE_{X_{t+1}^{*,p+2}\sim M_{t+1}(X_{t+1}^{*,p+1},\dd x_{t+1}^{*,p+2})}[\phi(X_{t+1}^{*,p+2})\mid X_{t+1}^{*,p+1}, X_{t+1}^{*,p}]\mid X_{t+1}^{*,p}]].
            \end{split}
        \end{equation}
        The second expectation is equal to
        \begin{equation}
            \label{eq:second_exp}
            \begin{split}
                &\bbE_{\bar{X}_{t+1}^{*,p+1}\sim Q_{t+1}^{\nu}(X_{t+1}^{*,p}, \dd \bar{x}_{t+1}^{*,p+1})}[
                \bbE_{X_{t+1}^{*,p+2}\sim M_{t+1}(\bar{X}_{t+1}^{*,p+1},\dd x_{t+1}^{*,p+2})}[\phi(X_{t+1}^{*,p+2})\mid \bar{X}_{t+1}^{*,p+1}]
                a_{t+1}^{\nu}(X_{t+1}^{*,p}, \bar{X}_{t+1}^{*,p+1})] \\
                &+ (1 -a_{t+1}^{\nu}(X_{t+1}^{*,p}, \bar{X}_{t+1}^{*,p+1})) \bbE_{X_{t+1}^{*,p+2}\sim M_{t+1}(X_{t+1}^{*,p},\dd x_{t+1}^{*,p+2})}[\phi(X_{t+1}^{*,p+2})\mid \bar{X}_{t+1}^{*,p}]
                ],
            \end{split}
        \end{equation}
        the first inner expectation in~\eqref{eq:second_exp} is estimated by
        \begin{equation}
            w_{t+1}(\bar{X}_{t}^{*,p+1}, \bar{X}_{t}^{*,p+2})\times( a_{t+1}(\bar{X}_{t}^{*,p+1}, \bar{X}_{t}^{*,p+2}) \phi(\bar{X}_{t}^{*,p+2}) + (1-a_{t+1}(\bar{X}_{t}^{*,p+1}, \bar{X}_{t}^{*,p+2}))\phi(\bar{X}_{t}^{*,p+1})),
        \end{equation}
        the second can be estimated similarly but replacing all the proposed particles $\bar{X}_{t}^{*,p+1}$ by $\tilde{X}_{t}^{*,p}$.
        The outer expectation in~\eqref{eq:second_exp} can be estimated using IS weights, and finally the outer expectation in~\eqref{eq:exp_order2} is estimated using $(W_t^{m,p}, X_t^{m,p})$.
    \end{proof}
\end{proposition}
